## 1. 这是什么？

这是一个**完整的图像生成工作流技能**，封装了从"想法"到"成品图"的全部环节：

```
你的想法 → 专业提示词 → AI生成图像 → 格式转换 → 保存到指定位置
```

### 核心特点

| 特点 | 说明 |
|------|------|
| **一键生图** | 说一句话，自动完成全部流程 |
| **专业提示词** | 自动将你的描述转为AI友好的英文提示词 |
| **格式自动转换** | 生成PNG后自动转JPG，苹果电脑直接打开 |
| **智能文件管理** | 自动保存到桌面/豆包AI生图文件夹 |
| **预设模板** | 内置多种风格模板，一键调用 |

## 2. 能做什么？

立即生成：清晨金色时刻、瀑布悬崖、能量团环绕、彩霞天空的灵性场景

### 5.2 核心代码

#### index.js

```javascript
const fs = require('fs');
const path = require('path');
const os = require('os');

// ★ API配置（从环境变量读取，隐藏）
const GEMINI_API_KEY = process.env.GEMINI_API_KEY;
const GEMINI_BASE_URL = process.env.GEMINI_BASE_URL || 'https://new.suxi.ai/v1beta';
const GEMINI_MODEL = 'gemini-3.1-flash-image-preview';
const DEFAULT_OUTPUT_DIR = path.join(os.homedir(), 'Desktop', '豆包AI生图');

// 预设模板
const PRESETS = {
  meditation_scene: {
    name: '灵性冥想场景',
    template: (custom) => `A serene spiritual awakening scene...
${custom ? '\nCustom: ' + custom : ''}`
  }
};

/**
 * 调用Gemini API生成图像
 */
async function callGeminiAPI(prompt) {
  if (!GEMINI_API_KEY) {
    throw new Error('GEMINI_API_KEY未配置');
  }
  
  const url = `${GEMINI_BASE_URL}/models/${GEMINI_MODEL}:generateContent?key=${GEMINI_API_KEY}`;
  
  const response = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      contents: [{ parts: [{ text: prompt }] }]
    })
  });
  
  if (!response.ok) {
    throw new Error(`API错误: ${response.status}`);
  }
  
  return response.json();
}

/**
 * 提取图像数据
 */
function extractImageFromResponse(data) {
  const parts = data.candidates?.[0]?.content?.parts || [];
  const imagePart = parts.find(p => p.inlineData);
  
  if (!imagePart) {
    return { success: false, error: '未生成图像' };
  }
  
  return {
    success: true,
    imageData: Buffer.from(imagePart.inlineData.data, 'base64')
  };
}

/**
 * PNG转JPG
 */
function saveAsJpg(imageData, outputPath) {
  const tmpPng = outputPath.replace('.jpg', '_tmp.png');
  fs.writeFileSync(tmpPng, imageData);
  
  try {
    const { execSync } = require('child_process');
    execSync(`python3 -c "from PIL import Image; img = Image.open('${tmpPng}'); 
      img = img.convert('RGB'); 
      img.save('${outputPath}', 'JPEG', quality=95)"`);
    fs.unlinkSync(tmpPng);
    return true;
  } catch (e) {
    fs.renameSync(tmpPng, outputPath.replace('.jpg', '.png'));
    return false;
  }
}

/**
 * 主工作流
 */
async function generateImageWorkflow({ description, preset, outputDir }) {
  // 确保目录存在
  const targetDir = outputDir || DEFAULT_OUTPUT_DIR;
  if (!fs.existsSync(targetDir)) {
    fs.mkdirSync(targetDir, { recursive: true });
  }
  
  // 获取提示词
  let finalPrompt = preset ? 
    PRESETS[preset].template('') : 
    `High quality: ${description}`;
  
  // 生成图像
  const apiResult = await callGeminiAPI(finalPrompt);
  const imageResult = extractImageFromResponse(apiResult);
  
  if (!imageResult.success) {
    return { success: false, error: imageResult.error };
  }
  
  // 保存文件
  const filename = `${Date.now()}_image.jpg`;
  const outputPath = path.join(targetDir, filename);
  
  const converted = saveAsJpg(imageResult.imageData, outputPath);
  
  return {
    success: true,
    path: converted ? outputPath : outputPath.replace('.jpg', '.png'),
    format: converted ? 'jpg' : 'png'
  };
}

module.exports = {
  name: 'huanxi-image-workflow',
  version: '1.0.0',
  tools: [{
    name: 'generate_image_workflow',
    description: '完整生图工作流',
    handler: generateImageWorkflow
  }]
};
```

## 7. 常见问题

### Q1: 生成的图片打不开？

**原因**：默认生成PNG格式，部分系统可能不兼容。

**解决**：本技能已自动转为JPG格式，无需额外操作。

### Q2: 提示"GEMINI_API_KEY未配置"？

**原因**：API密钥未设置。

**解决**：联系管理员获取密钥，添加到 `.env` 文件。

### Q3: 如何修改图片质量？

编辑代码中的 quality 参数：
```javascript
img.save('${outputPath}', 'JPEG', quality=95)  // 1-100
```

### Q4: 能生成其他尺寸吗？

目前Gemini模型自动决定尺寸。如需特定尺寸，可在提示词中说明：
```
生成16:9宽屏比例的图像：...
```
